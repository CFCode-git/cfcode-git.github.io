变量声明：let、const、var

Written by 周啟尧 with ♥ on December 28, 2019 in JavaScript 
a = 1
var a = 1
let a = 1
const a = 1

a = 1

var a // a1 全局变量
function fn(){
    var a // a2 局部变量
    function fn2(){
        a = 1  
        //  如果没有a1 a2，就声明一个全局变量a
        // 如果没有a1，就用a2
        // 如果没有a2.就用a1
        // 如果a1 a2都有，就近原则，用a2
    }
}

如果当前有a，就直接利用这个a，如果当前没有a，就声明一个全局变量a。

变量提升

function fn(){
    // {{a}}
    if(true){
        console.log(a)
    }else{
        // 这里不会执行
        var a = 1
        console.log(2)
    }
}
// undefined

代码没有报错，原因是var会变量提升。它最终会提到{{a}}的位置中去。

作用域

ES6以前

全局作用域、函数作用域

ES6

全局作用域、函数作用域、块级作用域

创建块级作用域

ES6以前

由于没有块级作用域，var声明的变量会有变量提升，因此需要**借助函数得到局部变量。**就是立即执行函数。

(function(){
    var a = 1  // 局部变量a
    window.y = function(){
        console.log(a)
    }
}()) 
// 匿名函数 立即调用

ES6

用’{}‘和let/const声明变量即可得到块级作用域。 块级作用域只能在块区域内被调用。

{
    let a = 1
    // const a = 1
}

let a = 1

let 作用域范围是离他最近的

{
    let a = 1
}
console.log(a)
// Uncaught ReferenceError: a is not defined

let 必须先声明后使用

{
    let a = 1
    {                     // 0
        console.log(a)    // 1
        let a = 2         // 2
        // 在这里log       // 3
        {
            let a = 3
        }
    }
}
// Uncaught ReferenceError: a is not defined
//

Temp Dead Zone

临时死区，理论上let a = 2的作用域是0-3，但实际上0-1这个区间内a是不允许使用的，因为没有声明。

let 不能重复声明同一个变量 可以赋值

{
    let a = 1
    console.log(a)
    let a = 2
    // 允许：a = 2 
    console.log(a)
}
// Uncaught SyntaxError :Identifier 'a' has already been declared

总结

  1. let 的作用域在最近的 {} 之间
  2. 如果在 let a之前使用a，报错
  3. 如果重复 let a，报错

const a = 1

与let一样，有两个区别： 4. 声明的时候必须赋值
5. 赋值后无法更改

for 循环中的 let 和 var

// var
for(var i=0; i<5; i++){
    setTimeout(()=>{
        console.log(i)
    },1000)
}
// 5
// 5
// 5
// 5
// 5

上述代码，由始至终都只有一个i，那就是全局作用域下的i。通过计时器，for循环执行完了之后才陆续打印出i的值，此时i的值已经变成了5。它相当于

var i=0;
// 满足i<5
i=1;
i=2;
i=3;
i=4;
i=5;
// 不满足i<5，退出循环
console.log(i)
console.log(i)
console.log(i)
console.log(i)
console.log(i)

// let
for(let i=0; i<5; i++){
    setTimeout(()=>{
        console.log(i)
    },1000)
}
// 0
// 1
// 2
// 3
// 4

上述代码，有6个i。()作用域中1个，然后每次循环js都帮我们在{}内重新声明一个新的i，这个新的i的值是()中i的值的拷贝。计时器每次打印的是{}作用域下的i。相当于

{
    let _i = 0; // i的拷贝，用_i表示，满足条件，执行
    console.log(_i)
}
{
    let _i = 1;
    console.log(_i)
}
{
    let _i = 2;
    console.log(_i)
}
{
    let _i = 3;
    console.log(_i)
}
{
    let _i = 4;
    console.log(_i)
}
i = 5 // 不满足，结束循环

理解实例

需求：点击第几项，就打印对应的n。

<!--html部分-->
<ul>
    <li>1</li>
    <li>2</li>
    <li>3</li>
    <li>4</li>
    <li>5</li>
    <li>6</li>
</ul>

//JavaScript
var liTags = document.querySelectorAll('li')

for(var i=0; i<liTags,length; i++){
    liTags[i].onclick = function(){
        //////////代码部分
    }
}

// 方法1 用 let 接收 i
for(var i=0; i<liTags,length; i++){
    let j = i
    liTags[j].onclick = function(){
        console.log(j)
    }
}

// 方法2 "let魔法"
for(let i=0; i<liTags,length; i++){
    liTags[i].onclick = function(){
        console.log(i)
    }
}
// 这里实际上js帮我们做了 "/ 1" 中的操作
//  i 的作用域只在圆括号()范围内
// js帮我们在{}中声明了一个新的i,是()中的i的拷贝(也叫i，用_i表示)
// 在{}的最后，将_i值又返还给()

// 方法3 立即执行函数
for(let i=0; i<liTags,length; i++){
    (function(j){
        liTags[j].onclick=function(){
            console.log(j)
        }
    })(i)
}
// 3 相当于
// 这里var j 不会跑出函数作用域
for(var i=0; i<5; i++){
    (function(){
        var j = arguments[0]
        liTags[j].onclick=function(){
            console.log(j)
        }
    })(i)
}

函数提升

函数声明会提升，函数表达式不会。

foo(); // "函数声明"
function foo(){
    console.log("函数声明")
}

foo2();  // typeError: foo2 is not a function
var foo2 = function(){
    console.log("函数表达式")
}
