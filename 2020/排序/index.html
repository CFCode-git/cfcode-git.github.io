四种排序算法

Written by 周啟尧 with ♥ on February 26, 2020 in

给出正整数数组 array = [2,1,5,3,8,4,9,5] 请写出一个函数 sort，使得 sort(array) 得到从小到大排好序的数组 [1,2,3,4,5,5,8,9] 新的数组可以是在 array 自身上改的，也可以是完全新开辟的内存。

不得使用 JS 内置的 sort API

快速排序

let quicksort = (numbers) => {
  if (numbers.length < 2) {
    return numbers;
  }
  let pivotindex = Math.floor(numbers.length / 2);
  let pivotnum = numbers.splice(pivotindex, 1)[0];
  let left = [];
  let right = [];
  for (i = 0; i < numbers.length; i++) {
    if (numbers[i] <= pivotnum) {
      left.push(numbers[i]);
    } else {
      right.push(numbers[i]);
    }
  }
  return quicksort(left).concat([pivotnum], quicksort(right));
};

选择排序

let sort = (numbers) => {
  if (numbers.length > 2) {
    let min_index = findmin(numbers);
    let min_num = numbers[min_index];
    numbers.splice(min_index, 1);
    return [min_num].concat(sort(numbers));
  } else {
    return numbers[0] < numbers[1] ? numbers : numbers.reverse();
  }
};

let findmin = (numbers) => {
  return (index = numbers.indexOf(min(numbers)));
};

let min = (numbers) => {
  if (numbers.length > 2) {
    return min([numbers[0], min(numbers.slice(1))]);
  } else {
    return Math.min.apply(null, numbers);
  }
};

let sort = (numbers) => {
  for (let i = 0; i < numbers.length - 1; i++) {
    let min_index = findmin(numbers.slice(i)) + i;
    if (min_index !== i) {
      swap(numbers, min_index, i);
    }
  }
  return numbers;
};

let findmin = (numbers) => {
  let index = 0;
  for (let i = 1; i < numbers.length; i++) {
    if (numbers[i] < numbers[index]) {
      index = i;
    }
  }
  return index;
};

let swap = (numbers, index, i) => {
  temp = numbers[index];
  numbers[index] = numbers[i];
  numbers[i] = temp;
};

计数排序

let countsort = (numbers) => {
  let hashtable = {},
    max = 0,
    result = [];
  for (i = 0; i < numbers.length; i++) {
    if (!(numbers[i] in hashtable)) {
      hashtable[numbers[i]] = 1;
    } else {
      hashtable[numbers[i]] += 1;
    }
    if (numbers[i] > max) {
      max = numbers[i];
    }
  }
  for (j = 0; j <= max; j++) {
    if (j in hashtable) {
      for (let k = 0; k < hashtable[j]; k++) {
        result.push(j);
      }
    }
  }
  return result;
};

归并排序

let margetsort = (numbers) => {
    let length = numbers.length
    if(length < 2){return numbers}
    let left = numbers.slice(0,Math.floor(length/2))
    let right = numbers.slice(Math.floor(length/2))
    return merge(mergesort(left),mergesort((right)))
}

let merge = (num1,num2) => {
    if(num1.length === 0){return num2}
    if(num2.length === 0){return num1}
    return num1[0] > num2[0] ?
        [num2[0]].concat(merge(num1,num2.slice(1))) : [num1[0]].concat(merge(num1.slice(1),num2))
}
