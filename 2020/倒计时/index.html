setTimeout 与 倒计时

Written by 周啟尧 with ♥ on April 25, 2020 in JavaScript

setInterval

function fn(countDown) {
  let time = countDown;
  let timeId = setInterval(() => {
    console.log(time);
    time = time - 1;
    if (time <= 0) {
      clearInterval(timeId);
    }
  }, 1000);
}

requestAnimationFrame

let a = 10;
let date = new Date();

let countDown = () => {
  if (new Date() - date < 1000) {
    requestAnimationFrame(countDown);
  } else {
    if (a > 0) {
      console.log(a--);
      date = new Date();
      requestAnimationFrame(countDown);
    }
  }
};

requestAnimationFrame(countDown);

setTimeout 实现 setInterval

const mySetInterval = (cb,time)=>{
  const fn = ()=>{
    cb()
    setTimeout(()=>{fn()},time)
  }
  setTimeout(fn,time)
}
