数组排序

给出正整数数组 array = [2,1,5,3,8,4,9,5] 请写出一个函数 sort，使得 sort(array) 得到从小到大排好序的数组 [1,2,3,4,5,5,8,9] 新的数组可以是在 array 自身上改的，也可以是完全新开辟的内存。

不得使用 JS 内置的 sort API

// 快速排序
function quicksort(array){
    if(array.length < 2){
        return array
    }
    let pivotIndex = Math.floor(array.length/2)
    let pivotNum = array.splice(pivotIndex,1)[0]
    let left = []
    let right = []
    for(i=0; i<array.length; i++){
        if(array[i]<=pivotNum){
            left.push(array[i])
        }else{
            right.push(array[i])
        }
    }
    return quicksort(left).concat([pivotNum],quicksort(right))
}
