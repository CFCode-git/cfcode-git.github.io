[面试题]数组去重

December 5, 2019

如何实现数组去重？ 假设有数组array = [1,5,2,3,4,2,3,1,3,4] 写一个unique函数去重

  1. 使用set实现

    function unique (arr) {
return Array.from(new Set(arr))
}

let array = [1,5,2,3,4,2,3,1,3,4]
console.log(unique(array))
// [1, 5, 2, 3, 4]

缺点：这个方法不能去掉重复的空数组{}。

var arr = [1,1,'true','true',true,true,15,15,false,false, undefined,undefined, null,null, NaN, NaN,'NaN', 0, 0, 'a', 'a',{},{}];
console.log(unique(arr))

// [1, "true", true, 15, false, undefined, null, NaN, "NaN", 0, "a", {}, {}]

  1. 不使用set –利用Map数据结构

    function unique(array){
let hashMap = new Map() // 用于判别是否重复
let result = new Array() // 用于返回结果
for (let i=0; i<array.length; i++){
    if(hashMap.has(array[i])){
        hashMap.set(array[i],true) // Map中已经有这个key值，标记为true
    }else{
        hashMap.set(array[i],false) // Map中没有这个key，将这个值push到结果，并在hashMap中标记为false
        result.push(array[i])
    }
}
return result
}

    优点：可以找到重复的值，hashMap中值为true的值为重复的值。

  2. 使用双层循环 –splice去重

    function unique(array){
for (let i=0; i<array.length; i++){
    for(let j=i+1; j<array.length; j++){ // 从 i 的下一项开始找有没有跟 i 一样的
        if(array[i]===array[j]){
            array.splice(j,1); // 找到一样的 删掉当前的 j
            j -=1; // j 后面的元素往前移动，j 不要加1，再次从当前的 j 继续往后遍历。
        }
    }
}
return array
}

  3. 利用indexOf去重

    function unique(array){
let arr = []
for(let i=0; i<array.length; i++){ 
    if(arr.indexOf(array[i]) === -1){ // 如果空数组里面没有这个值，就push
        arr.push(array[i])
    }
}
return arr
}
